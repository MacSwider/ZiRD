CREATE OR REPLACE FUNCTION get_job_title (p_job_id IN VARCHAR2)
RETURN VARCHAR2
IS
    v_job_title jobs.job_title%TYPE;
BEGIN
    SELECT job_title INTO v_job_title
    FROM jobs
    WHERE job_id = p_job_id;
    
    RETURN v_job_title;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(-20001, 'Praca o podanym ID (' || p_job_id || ') nie istnieje.');
END;
/

CREATE OR REPLACE FUNCTION get_annual_salary (p_emp_id IN NUMBER)
RETURN NUMBER
IS
    v_salary employees.salary%TYPE;
    v_comm employees.commission_pct%TYPE;
    v_annual_salary NUMBER;
BEGIN
    SELECT salary, commission_pct INTO v_salary, v_comm
    FROM employees
    WHERE employee_id = p_emp_id;
    
    v_annual_salary := (v_salary * 12) + (v_salary * NVL(v_comm, 0));
    
    RETURN v_annual_salary;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN NULL;
END;
/

CREATE OR REPLACE FUNCTION format_phone_number (p_phone IN VARCHAR2)
RETURN VARCHAR2
IS
    v_formatted_phone VARCHAR2(100);
BEGIN
    IF INSTR(p_phone, '.') > 0 THEN
        v_formatted_phone := REGEXP_REPLACE(p_phone, '^([0-9]{3})\.', '(\1) ');
    ELSE
        v_formatted_phone := '(' || SUBSTR(p_phone, 1, 2) || ') ' || SUBSTR(p_phone, 3);
    END IF;
    
    RETURN v_formatted_phone;
END;
/

CREATE OR REPLACE FUNCTION format_string_edges (p_str IN VARCHAR2)
RETURN VARCHAR2
IS
    v_len NUMBER := LENGTH(p_str);
BEGIN
    IF v_len IS NULL OR v_len = 0 THEN
        RETURN p_str;
    ELSIF v_len = 1 THEN
        RETURN UPPER(p_str);
    ELSE
        RETURN UPPER(SUBSTR(p_str, 1, 1)) ||
                LOWER(SUBSTR(p_str,2, v_len - 2)) ||
                UPPER(SUBSTR(p_str, v_len, 1));
    END IF;
END;
/

CREATE OR REPLACE FUNCTION pesel_to_date (p_pesel IN VARCHAR2)
RETURN VARCHAR2
IS
    v_year NUMBER;
    v_month NUMBER;
    v_day VARCHAR2(2);
    v_century VARCHAR2(2);
    v_final_date DATE;
BEGIN
    IF LENGTH(p_pesel) != 11 THEN
        RAISE_APPLICATION_ERROR(-20002, 'Nieprawidłowa długość nr PESEL.');
    END IF;
    
    v_year := TO_NUMBER(SUBSTR(p_pesel, 1, 2));
    v_month := TO_NUMBER(SUBSTR(p_pesel, 3, 2));
    v_day := SUBSTR(p_pesel, 5, 2);
    
    IF v_month BETWEEN 81 AND 92 THEN
        v_century := '18';
        v_month := v_month - 80;
    ELSIF v_month BETWEEN 1 AND 12 THEN
        v_century := '19';
    ELSIF v_month BETWEEN 21 AND 32 THEN
        v_century := '20';
        v_month := v_month - 20;
    ELSIF v_month BETWEEN 41 AND 52 THEN
        v_century := '21';
        v_month := v_month - 40;
    ELSIF v_month BETWEEN 61 AND 72 THEN
        v_century := '22';
        v_month := v_month - 60;
    END IF;
    
    v_final_date := TO_DATE(v_century || TO_CHAR(v_year, 'FM00') || '-' || TO_CHAR(v_month, 'FM00') || '-' || v_day, 'YYYY-MM-DD');
    
    RETURN TO_CHAR(v_final_date, 'YYYY-MM-DD');
END;
/

CREATE OR REPLACE FUNCTION get_country_stats (p_country_name IN VARCHAR2)
RETURN VARCHAR2
IS
    v_country_exists NUMBER;
    v_emp_count NUMBER;
    v_dept_count NUMBER;
BEGIN
    SELECT COUNT(*) INTO v_country_exists
    FROM countries
    WHERE UPPER(country_name) = UPPER(p_country_name);
    
    IF v_country_exists = 0 THEN
        RAISE_APPLICATION_ERROR(-20003, 'KRAJ O PODANEJ NAZWIE NIE ISTNIEJE W BAZIE.');
    END IF;
    
    SELECT COUNT(e.employee_id), COUNT(DISTINCT d.department_id)
    INTO v_emp_count, v_dept_count
    FROM countries c
    JOIN locations l ON c.country_id = l.country_id
    JOIN departments d ON l.location_id = d.location_id
    LEFT JOIN employees e ON d.department_id = e.department_id
    WHERE UPPER(c.country_name) = UPPER(p_country_name);
    
    RETURN 'Pracownicy: ' || v_emp_count || ', Departmenty: ' || v_dept_count;
END;
/

CREATE TABLE archiwum_departamentow (
    id NUMBER,
    nazwa VARCHAR2(30),
    data_zamkniecia DATE,
    ostatni_manager VARCHAR2(100)
);

CREATE OR REPLACE TRIGGER trg_archive_dept
AFTER DELETE ON departments
FOR EACH ROW
DECLARE
    v_manager_name VARCHAR2(100) := 'Brak Managera';
BEGIN
    IF :OLD.manager_id IS NOT NULL THEN
        BEGIN
            SELECT first_name || ' ' || last_name
            INTO v_manager_name
            FROM employees
            WHERE employee_id = :OLD.manager_id;
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                v_manager_name := 'Nieznany (Usunięty pracownik)';
        END;
    END IF;
    
    INSERT INTO archiwum_departamentow (id, nazwa, data_zamkniecia, ostatni_manager)
    VALUES (:OLD.department_id, :OLD.department_name, SYSDATE, v_manager_name);
END;
/

CREATE TABLE zlodziej (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    uzytkownik VARCHAR2(50),
    czas_zmiany DATE
);

CREATE OR REPLACE TRIGGER trg_check_salary
BEFORE INSERT OR UPDATE OF salary ON employees
FOR EACH ROW
DECLARE
    PRAGMA AUTONOMOUS_TRANSACTION; 
BEGIN
    IF :NEW.salary NOT BETWEEN 2000 AND 26000 THEN
        INSERT INTO zlodziej (uzytkownik, czas_zmiany) 
        VALUES (USER, SYSDATE);
        
        COMMIT;
        
        RAISE_APPLICATION_ERROR(-20004, 'Wynagrodzenie poza widełkami 2000-26000!');
    END IF;
END;
/

CREATE SEQUENCE seq_employees_id START WITH 1000 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER trg_emp_autoincrement
BEFORE INSERT ON employees
FOR EACH ROW
BEGIN
    IF :NEW.employee_id IS NULL THEN
        :NEW.employee_id := seq_employees_id.NEXTVAL;
    END IF;
END;
/

CREATE OR REPLACE TRIGGER trg_lock_job_grades
BEFORE INSERT OR UPDATE OR DELETE ON job_grades
BEGIN
    RAISE_APPLICATION_ERROR(-20005, 'Tabela JOB_GRADES jest zablokowana do odczytu. Operacje DML są zabronione.');
END;
/

CREATE OR REPLACE TRIGGER trg_protect_jobs_salary
BEFORE UPDATE ON jobs
FOR EACH ROW
BEGIN
    :NEW.min_salary := :OLD.min_salary;
    :NEW.max_salary := :OLD.max_salary;
END;
/

CREATE OR REPLACE PACKAGE pkg_regions_crud IS
    PROCEDURE add_region(p_region_id IN NUMBER, p_region_name IN VARCHAR2);
    FUNCTION get_region_name(p_region_id IN NUMBER) RETURN VARCHAR2;
    FUNCTION get_all_regions RETURN SYS_REFCURSOR;
    PROCEDURE update_region(p_region_id IN NUMBER, p_new_region_name IN VARCHAR2);
    PROCEDURE delete_region(p_region_id IN NUMBER);
END pkg_regions_crud;
/

CREATE OR REPLACE PACKAGE BODY pkg_regions_crud IS
    PROCEDURE add_region(p_region_id IN NUMBER, p_region_name IN VARCHAR2) IS
    BEGIN
        INSERT INTO regions(region_id, region_name) 
        VALUES (p_region_id, p_region_name);
        COMMIT;
    END add_region;

    FUNCTION get_region_name(p_region_id IN NUMBER) RETURN VARCHAR2 IS
        v_name regions.region_name%TYPE;
    BEGIN
        SELECT region_name INTO v_name 
        FROM regions 
        WHERE region_id = p_region_id;
        
        RETURN v_name;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 'Brak regionu o ID ' || p_region_id;
    END get_region_name;

    FUNCTION get_all_regions RETURN SYS_REFCURSOR IS
        v_cursor SYS_REFCURSOR;
    BEGIN
        OPEN v_cursor FOR SELECT * FROM regions ORDER BY region_id;
        RETURN v_cursor;
    END get_all_regions;

    PROCEDURE update_region(p_region_id IN NUMBER, p_new_region_name IN VARCHAR2) IS
    BEGIN
        UPDATE regions 
        SET region_name = p_new_region_name 
        WHERE region_id = p_region_id;
        COMMIT;
    END update_region;

    PROCEDURE delete_region(p_region_id IN NUMBER) IS
    BEGIN
        DELETE FROM regions 
        WHERE region_id = p_region_id;
        COMMIT;
    END delete_region;

END pkg_regions_crud;
/
